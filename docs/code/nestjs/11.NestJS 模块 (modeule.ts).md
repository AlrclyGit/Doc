# 模块 (modeule.ts)

```typescript
import { DynamicModule, Global, Module } from '@nestjs/common'

@Global() // 全局模块装饰器
@Module({
  imports: [LoginModule],
  providers: [LoginService],
  controllers: [LoginController],
  exports: [LoginModule],
})
export class ConfigModule {
  // 动态 module
  static forRoot(options: string): DynamicModule {
    return {
      module: ConfigModule,
      providers: [
        {
          provide: 'config',
          useValue: { baseUrl: `/api${options}` },
        },
      ],
    }
  }
}
```
