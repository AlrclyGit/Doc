# 组件化开发

## 非构建项目的全局组件

```html
<body>
    <div id="app">
        <div>
            {{ tip }}
            <af-title />
            <af-content />
        </div>
    </div>
    <script type="text/x-template" id="afTitle">
        <h1>Good Morning</h1>
        <af-content/>
    </script>
    <script type="text/x-template" id="afContent">
        <p>Good a nice day</p>
    </script>
    <script>
        const app = Vue.createApp({
            data() {
                // 数据
                return {
                    tip: 'This is Test Page',
                }
            },
        })
        app.component('afTitle', {
            template: '#afTitle',
        })
        app.component('afContent', {
            template: '#afContent',
        })
        app.mount('#app')
    </script>
</body>
```

## 非构建项目的局部组件

```html
<body>
    <div id="app">
        <div>
            {{ tip }}
            <af-title />
        </div>
    </div>
    <script type="text/x-template" id="afTitle">
        <h1>Good Morning</h1>
        <af-content/>
    </script>
    <script type="text/x-template" id="afContent">
        <p>Good a nice day</p>
    </script>
    <script>
        const afContent = {
            data() {
                // 数据
                return {
                    tip: 'This is afContent Page',
                }
            },
            template: '#afContent',
        }
        const afTitle = {
            data() {
                // 数据
                return {
                    tip: 'This is afTitle Page',
                }
            },
            template: '#afTitle',
            components: { afContent },
        }
        const app = Vue.createApp({
            data() {
                // 数据
                return {
                    tip: 'This is Test Page',
                }
            },
            components: { afTitle },
        })
        app.mount('#app')
    </script>
</body>
```

## 使用 ES 模块构建 VUE3 项目

### main.js

```javascript
import App from './App.js'
export default Vue.createApp(App).mount('#app')
```

### App.js

```javascript
import Todo from './components/Todo.js'
export default {
    components: { Todo },
    data() {
        return {
            title: 'Good morning',
            db: [
                { id: 1, title: '语文' },
                { id: 2, title: '数学' },
                { id: 3, title: '英文' },
            ],
        }
    },
}
```

### index.html

```html
<!DOCTYPE html>
<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>Document</title>
        <script src="https://cdn.bootcdn.net/ajax/libs/vue/3.3.4/vue.global.js"></script>
    </head>

    <body>
        <div id="app">
            <template v-for="data in db" :key="data.id">
                <todo :data="data" />
            </template>
        </div>
        <script type="module">
            import main from './main.js'
        </script>
    </body>
</html>
```

### components/Todo.js

```javascript
export default {
    props: ['data'],
    template: '<div>{{data.title}}</div>',
}
```
